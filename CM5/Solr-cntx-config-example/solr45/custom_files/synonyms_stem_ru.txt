# Словарь синонимов (ru) для использования совместно со stemming-фильтром

# Примеры описания синонимов
# couch,sofa,divan
# huge,ginormous,humungous => large
# small => tiny,teeny,weeny

# Поскольку stemming  обрезает слова до некоторого остова (он не всегда совпадает с корнем слова), 
# то и данный словарь должен быть составлен из таких остовов, например, если нужно, чтобы 
# слова "день", "ночь" и "сутки" обрабатывались при поиске как синонимы, то в словаре они должны быть 
# предствалены следующим образом   
# ноч,ден,сутк  
сноумен => снеговик

# пример исправления опечаток
сенг => снег
